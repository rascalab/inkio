# TocBlock — Inline Table of Contents Extension

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a Notion-style inline ToC block that lives inside the document, auto-lists headings, and scrolls on click.

**Architecture:** A Tiptap Node extension (`TocBlock`) with a React NodeView that reads headings from the document via `getHeadingsFromContent`. The block is leaf (no editable content), inserted via slash command `/toc`. It subscribes to editor `update` events to stay in sync. CSS goes in `core/style.css` + `core/minimal.css`. The existing external `<ToC>` component stays as-is.

**Tech Stack:** Tiptap Node extension, React NodeView, existing `getHeadingsFromContent` from `ToC.ts`

### Task 1: TocBlock Node Extension

**Files:**
- Create: `packages/core/src/extensions/TocBlock/TocBlock.ts`

- [ ] **Step 1: Create the extension**

```ts
import { Node } from '@tiptap/core';
import { ReactNodeViewRenderer } from '@tiptap/react';
import { TocBlockView } from './TocBlockView';

export interface TocBlockOptions {
  maxLevel: number;
}

export const TocBlock = Node.create<TocBlockOptions>({
  name: 'tocBlock',
  group: 'block',
  atom: true,

  addOptions() {
    return { maxLevel: 3 };
  },

  addAttributes() {
    return {
      maxLevel: { default: 3, rendered: false },
    };
  },

  parseHTML() {
    return [{ tag: 'div[data-type="toc"]' }];
  },

  renderHTML() {
    return ['div', { 'data-type': 'toc' }, 0];
  },

  addNodeView() {
    return ReactNodeViewRenderer(TocBlockView);
  },
});
```

### Task 2: TocBlock React NodeView

**Files:**
- Create: `packages/core/src/extensions/TocBlock/TocBlockView.tsx`

- [ ] **Step 1: Create the NodeView component**

```tsx
import { type NodeViewProps, NodeViewWrapper } from '@tiptap/react';
import { useEffect, useMemo, useState } from 'react';
import { getHeadingsFromContent, type HeadingItem } from '../../components/ToC';

export function TocBlockView({ editor, node }: NodeViewProps) {
  const maxLevel = (node.attrs.maxLevel as number) || 3;
  const [headings, setHeadings] = useState<HeadingItem[]>([]);

  useEffect(() => {
    const update = () => {
      setHeadings(getHeadingsFromContent(editor.getJSON()));
    };
    update();
    editor.on('update', update);
    return () => { editor.off('update', update); };
  }, [editor]);

  const filtered = useMemo(
    () => headings.filter((h) => h.level <= maxLevel),
    [headings, maxLevel],
  );

  const handleClick = (e: React.MouseEvent, index: number) => {
    e.preventDefault();
    const els = Array.from(
      editor.view.dom.querySelectorAll('h1, h2, h3, h4, h5, h6'),
    ).filter((el) => parseInt(el.tagName[1], 10) <= maxLevel);
    els[index]?.scrollIntoView({ behavior: 'smooth', block: 'start' });
  };

  if (filtered.length === 0) {
    return (
      <NodeViewWrapper className="inkio-toc-block" contentEditable={false}>
        <p className="inkio-toc-block-empty">Add headings to see the table of contents.</p>
      </NodeViewWrapper>
    );
  }

  const minLevel = Math.min(...filtered.map((h) => h.level));

  return (
    <NodeViewWrapper className="inkio-toc-block" contentEditable={false}>
      <ul className="inkio-toc-block-list">
        {filtered.map((heading, i) => (
          <li
            key={`${heading.id}-${i}`}
            className="inkio-toc-block-item"
            style={{ '--inkio-toc-depth': heading.level - minLevel } as React.CSSProperties}
          >
            <a
              href={`#${heading.id}`}
              className="inkio-toc-block-link"
              onClick={(e) => handleClick(e, i)}
            >
              {heading.text}
            </a>
          </li>
        ))}
      </ul>
    </NodeViewWrapper>
  );
}
```
